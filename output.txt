function _<(model: Model, action: Action) => Model>(): (model: Model, action: Action) => Model;
type Model = [Grid, string, string[]];
type Grid = string[][];
type Emoji = string;
type Action = SelectEmoji | StampEmoji | ClearCell | ClearGrid | FillRow;
type SelectEmoji = {    type: "SelectEmoji";    emoji: Emoji;};
type StampEmoji = {    type: "StampEmoji";    row: Row;    col: Col;};
type Row = number;
type Col = number;
type ClearCell = {    type: "ClearCell";    row: Row;    col: Col;};
type ClearGrid = {    type: "ClearGrid";};
type FillRow = {    type: "FillRow";    row: Row;};
